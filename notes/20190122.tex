\begin{theorem}
  \textbf{(Existence of minima and maxima).} Let $C\subset \R^n$ be a compact set, and let $f : C\rightarrow \R$ be a continuous function. Then there exists a point $\bm{a}\in C$ such that $f(\bm{a})\geq f(\bm{x})$ for all $\bm{x}\in C$, and a point $\bm{b}\in C$ such that $f(\bm{b})\leq f(\bm{x})$ for all $\bm{x}\in C$.
